package org.example;  
  
public class MyTread extends Thread{  
    static int ticket=0;  
    static Object lock=new Object();  
    @Override  
    public void run(){  
        while(true){  
        //不能变到上面因为........................。
            synchronized(lock){  
                if(ticket<100){  
                    try {  
                        Thread.sleep(10);  
                    } catch (InterruptedException e) {  
                        throw new RuntimeException(e);  
                    }  
                    ticket++;  
                    System.out.println(("正在买第"+ticket+"张票"));  
                }else{  
                    break;  
                    }  }  }  }  }
